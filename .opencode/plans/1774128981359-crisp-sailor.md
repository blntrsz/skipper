```typescript
import { Effect } from "effect";
import { SandboxService } from "../port/sandbox.service";
import { FileSystemService } from "../port/file-system.service";
import type { ProjectModel } from "../domain/project.model";
import { ChildProcess } from "effect/unstable/process";

export const runCommandInWorkspace = Effect.fn("workspace.runCommand")(function* (
  projectModel: ProjectModel,
  command: string,
) {
  const sandbox = yield* SandboxService;
  const fileSystem = yield* FileSystemService;

  const cwd = projectModel.branch
    ? yield* fileSystem.branchProjectCwd(projectModel)
    : yield* fileSystem.mainProjectCwd(projectModel);

  const cmd = ChildProcess.make({
    stdin: "inherit",
    stdout: "inherit",
    stderr: "inherit",
  })`bash -c ${command}`;

  yield* sandbox.execute(cmd.pipe(ChildProcess.setCwd(cwd)));
});
```

### 3. `packages/cli/src/command/sandbox/run-sandbox.command.ts`

```typescript
import { Workspace } from "@skippercorp/core";
import { Effect } from "effect";
import { Command, Flag, Prompt } from "effect/unstable/cli";
import { localWorkTreeLayer } from "@skippercorp/core/runtime";
import { flags, pickProject } from "./sandbox.common";

export const runSandboxCommand = Command.make(
  "run",
  {
    ...flags,
    command: Flag.optional(
      Flag.string("command").pipe(Flag.withAlias("cmd"), Flag.withDescription("Bash command")),
    ),
  },
  (config) =>
    Effect.gen(function* () {
      const project = yield* pickProject(config.git, false);

      const command = yield* Option.match(config.command, {
        onSome: (value) => Effect.succeed(value),
        onNone: () =>
          Prompt.run(
            Prompt.text({
              message: "Command",
              validate: (value) =>
                value.trim().length > 0
                  ? Effect.succeed(value)
                  : Effect.fail("Command is required"),
            }),
          ),
      });

      yield* Workspace.runCommandInWorkspace(project, command);
    }).pipe(Effect.provide(localWorkTreeLayer)),
).pipe(Command.withDescription("Run a command in a sandbox workspace"));
```
