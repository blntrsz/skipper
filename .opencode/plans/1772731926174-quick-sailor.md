# Task Domain Implementation Plan

## Overview
Create a Task domain for tracking agent tasks across multiple Git repositories with SQLite persistence using Effect SQL patterns.

## Requirements
- **Task entity**: id (ulid), description (immutable), repository (GitRepository), state (working | asking-question | done | done-and-seen), createdAt, updatedAt
- **Storage**: SQLite at `${homeDir}/.local/share/skipper/skipper.db` via @effect/sql-sqlite-bun
- **State transitions**: Free (any state → any state)
- **Migrations**: Separate migration files using Migrator

## File Structure

```

## Implementation Details

### 1. Domain Models (Single File)

#### `/src/effect/domain/Task.ts`
All domain types in one file using Schema.Class:

```typescript
import { Schema } from "effect";
import { ulid } from "ulid";

// TaskId - Branded ULID
export const TaskId = Schema.String.pipe(Schema.brand("TaskId"));
export type TaskId = typeof TaskId.Type;

export function generateTaskId(): TaskId {
  return TaskId.makeUnsafe(ulid());
}

// TaskState - Literal union
export const TaskState = Schema.Literal(
  "working",
  "asking-question",
  "done",
  "done-and-seen"
);
export type TaskState = typeof TaskState.Type;

// Task model using Schema.Class
export class Task extends Schema.Class<Task>("Task")({
  id: TaskId,
  description: Schema.String,
  repository: Schema.String,  // JSON stringified GitRepository
  state: TaskState,
  createdAt: Schema.Number,
  updatedAt: Schema.Number,
}) {}

// DTOs - picked from Task
export const TaskCreate = Task.pick("description", "repository");
export type TaskCreate = typeof TaskCreate.Type;

export const TaskUpdate = Task.pick("state");
export type TaskUpdate = typeof TaskUpdate.Type;

// Error - part of domain
export class TaskNotFoundError extends Schema.TaggedErrorClass<TaskNotFoundError>(
  "TaskNotFoundError"
)("TaskNotFoundError", {
  id: TaskId,
}) {
  get message() {
    return `Task not found: ${this.id}`;
  }
}
```

**Note**: Store GitRepository as JSON string in DB, parse when needed.

### 2. Database Layer

#### `/src/effect/internal/DatabaseService.ts`
Use @effect/sql-sqlite-bun with FileSystem service:

```typescript
import { Effect, Layer } from "effect";
import { FileSystem } from "@effect/platform";
import { BunSqliteClient } from "@effect/sql-sqlite-bun";
import { BunContext } from "@effect/platform-bun";
import { Migrator } from "@effect/sql";
import { homedir } from "node:os";
import { join } from "node:path";
import { fileURLToPath } from "node:url";

const DB_PATH = join(homedir(), ".local/share/skipper/skipper.db");
const DB_DIR = join(homedir(), ".local/share/skipper");

// Ensure directory exists using FileSystem
const ensureDir = Effect.gen(function* () {
  const fs = yield* FileSystem.FileSystem;
  const exists = yield* fs.exists(DB_DIR);
  if (!exists) {
    yield* fs.makeDirectory(DB_DIR, { recursive: true });
  }
});

// SQLite client layer
export const DatabaseLive = BunSqliteClient.layer({
  filename: DB_PATH,
});

// Migrator layer
export const MigratorLive = Migrator.layer({
  loader: Migrator.fromFileSystem(
    fileURLToPath(new URL("../migrations", import.meta.url))
  ),
  schemaDirectory: "src/migrations",
}).pipe(Layer.provide(DatabaseLive));

// Combined layer with directory setup
export const SqlLive = Layer.unwrapEffect(
  Effect.gen(function* () {
    yield* ensureDir;
    return Layer.mergeAll(DatabaseLive, MigratorLive);
  })
).pipe(Layer.provide(BunContext.layer));
```

**Database Schema** (created via migrations):
```sql
CREATE TABLE IF NOT EXISTS tasks (
  id TEXT PRIMARY KEY,
  description TEXT NOT NULL,
  repository TEXT NOT NULL,  -- JSON stringified GitRepository
  state TEXT NOT NULL,
  created_at INTEGER NOT NULL,
  updated_at INTEGER NOT NULL
);

CREATE INDEX IF NOT EXISTS idx_tasks_repository ON tasks(repository);
CREATE INDEX IF NOT EXISTS idx_tasks_state ON tasks(state);
```

### 4. Service Layer

#### `/src/effect/Task/Service.ts`
Service + SqlModel repository:

```typescript
import { Effect, Layer } from "effect";
import { SqlClient } from "@effect/sql";
import * as SqlModel from "@effect/sql/SqlModel";
import { Task, TaskCreate, TaskUpdate, generateTaskId, TaskNotFoundError } from "../domain/Task";

// Task model for SqlModel
const TaskModel = SqlModel.make(Task, {
  idColumn: "id",
  tableName: "tasks",
  spanPrefix: "TaskRepository",
  // Map domain <-> database (column names)
  decode: (row) => Task.make({
    id: row.id,
    description: row.description,
    repository: row.repository,
    state: row.state,
    createdAt: row.created_at,
    updatedAt: row.updated_at,
  }),
  encode: (task) => ({
    id: task.id,
    description: task.description,
    repository: task.repository,
    state: task.state,
    created_at: task.createdAt,
    updated_at: task.updatedAt,
  }),
});

// Repository from SqlModel
const TaskRepository = SqlModel.makeRepository(TaskModel, {
  tableName: "tasks",
  spanPrefix: "TaskRepository",
  idColumn: "id",
});

// Task Service
export const TaskService = ServiceMap.Service<{
  createTask: (data: TaskCreate) => Effect.Effect<Task, never, SqlClient.SqlClient>;
  getTask: (id: TaskId) => Effect.Effect<Task, TaskNotFoundError, SqlClient.SqlClient>;
  getTasksByRepository: (repo: string) => Effect.Effect<ReadonlyArray<Task>, never, SqlClient.SqlClient>;
  getAllTasks: () => Effect.Effect<ReadonlyArray<Task>, never, SqlClient.SqlClient>;
  updateTaskState: (id: TaskId, state: TaskState) => Effect.Effect<Task, TaskNotFoundError, SqlClient.SqlClient>;
  deleteTask: (id: TaskId) => Effect.Effect<void, TaskNotFoundError, SqlClient.SqlClient>;
}>("TaskService");

// Implementation
export const TaskServiceLive = Layer.effect(
  TaskService,
  Effect.gen(function* () {
    const sql = yield* SqlClient.SqlClient;
    const repo = yield* TaskRepository;

    return TaskService.of({
      createTask: (data) =>
        Effect.gen(function* () {
          const now = Date.now();
          const task = yield* repo.insert({
            id: generateTaskId(),
            ...data,
            state: "working",
            createdAt: now,
            updatedAt: now,
          });
          return task;
        }),

      getTask: (id) =>
        Effect.gen(function* () {
          const task = yield* repo.findById(id);
          if (!task) {
            yield* Effect.fail(new TaskNotFoundError({ id }));
          }
          return task;
        }),

      getTasksByRepository: (repo) =>
        Effect.gen(function* () {
          const tasks = yield* sql`SELECT * FROM tasks WHERE repository LIKE ${"%" + repo + "%"}`;
          return tasks.map(TaskModel.decode);
        }),

      getAllTasks: () => repo.findAll(),

      updateTaskState: (id, state) =>
        Effect.gen(function* () {
          const now = Date.now();
          const task = yield* repo.update({ id, state, updatedAt: now });
          return task;
        }),

      deleteTask: (id) => repo.delete(id),
    });
  })
);
